async function doLogin(){
  const emailRaw=$('loginEmail').value.trim();
  const email=emailRaw.toLowerCase();
  const password=$('loginPassword').value;
  $('loginError').classList.add('hidden');

  if(!emailRaw||!password){
    $('loginError').textContent='יש למלא מייל וסיסמה';
    $('loginError').classList.remove('hidden');
    return;
  }

  if(!email.endsWith(ALLOWED_EMAIL_DOMAIN)){
    $('loginError').textContent='הכניסה מותרת רק למייל שמסתיים ב־'+ALLOWED_EMAIL_DOMAIN;
    $('loginError').classList.remove('hidden');
    return;
  }

  if(password!==SHARED_LOGIN_PASSWORD){
    $('loginError').textContent='סיסמה שגויה';
    $('loginError').classList.remove('hidden');
    return;
  }

  $('loginBtn').disabled=true;
  $('loginBtn').textContent='מתחבר...';

  try{
    // 1) ניסיון התחברות רגילה ל-Supabase Auth
    let res=await fetch(`${SUPABASE_URL}/auth/v1/token?grant_type=password`,{
      method:'POST',
      headers:{'apikey':SUPABASE_KEY,'Content-Type':'application/json'},
      body:JSON.stringify({email:emailRaw,password:SHARED_LOGIN_PASSWORD})
    });
    let data=await res.json();

    // 2) אם המשתמש עדיין לא קיים, ננסה ליצור אותו אוטומטית
    if(!data.access_token){
      const signup=await fetch(`${SUPABASE_URL}/auth/v1/signup`,{
        method:'POST',
        headers:{'apikey':SUPABASE_KEY,'Content-Type':'application/json'},
        body:JSON.stringify({email:emailRaw,password:SHARED_LOGIN_PASSWORD})
      });
      const signupData=await signup.json();

      // אם Supabase מחזיר session אחרי signUp
      if(signupData.access_token){
        data=signupData;
      }else if(signupData.user && !signupData.access_token){
        throw new Error('המשתמש נוצר אבל צריך אישור מייל / Confirm Email ב-Supabase לפני כניסה.');
      }else{
        throw new Error(signupData.error_description || signupData.msg || signupData.error || data.error_description || data.msg || data.error || 'שגיאה בהתחברות');
      }
    }

    accessToken=data.access_token;
    currentUser=data.user || {email:emailRaw};
    localStorage.setItem('mashav_session',JSON.stringify({
      access_token:data.access_token,
      refresh_token:data.refresh_token,
      expires_at:data.expires_at,
      user:data.user || {email:emailRaw}
    }));

    await showApp();

  }catch(e){
    $('loginError').textContent='שגיאה: '+e.message;
    $('loginError').classList.remove('hidden');
  }finally{
    $('loginBtn').disabled=false;
    $('loginBtn').textContent='🔐 התחבר למערכת';
  }
}

async function refreshSession(refreshToken){
  try{
    const res=await fetch(`${SUPABASE_URL}/auth/v1/token?grant_type=refresh_token`,{
      method:'POST',
      headers:{'apikey':SUPABASE_KEY,'Content-Type':'application/json'},
      body:JSON.stringify({refresh_token:refreshToken})
    });
    const data=await res.json();
    const email=(data.user?.email||'').toLowerCase();
    if(data.access_token && email.endsWith(ALLOWED_EMAIL_DOMAIN)){
      accessToken=data.access_token;
      currentUser=data.user;
      localStorage.setItem('mashav_session',JSON.stringify({
        access_token:data.access_token,
        refresh_token:data.refresh_token,
        expires_at:data.expires_at,
        user:data.user
      }));
      await showApp();
      return true;
    }
  }catch{}
  return false
}

  loadLocal();
  const saved=localStorage.getItem('mashav_session');
  if(saved){
    try{
      const s=JSON.parse(saved);
      const email=(s.user?.email||'').toLowerCase();

          currentUser=s.user;
          await showApp();

      }
    }catch{}
  }
  $('loginEmail').value=DEFAULT_LOGIN_EMAIL;
  $('loginScreen').classList.remove('hidden')
}

async function saveLog(silent=false,fromAuto=false,forceNew=false){
  let log=formLog();
  if(!log.contractor_name){toast('חובה למלא שם קבלן','bad');return}
  if(!log.project_number||!log.project_name){toast('חובה למלא פרויקט','bad');return}

  const existing=editingId ? logs.find(x=>x.id===editingId) : null;

  // אם המשתמש שינה קבלן/חודש/פרויקט בזמן שערך יום קיים,
  // לא מחליפים את הרשומה הקודמת בלי לשאול — כדי לא למחוק קבלן אחר בטעות.
  if(existing && !fromAuto && !forceNew){
    const identityChanged =
      existing.contractor_name!==log.contractor_name ||
      existing.month_key!==log.month_key ||
      existing.project_number!==log.project_number;
    if(identityChanged){
      const createNew=confirm('שינית קבלן / חודש / פרויקט. ליצור יום עבודה חדש במקום להחליף את הקיים?\\n\\nאישור = ליצור חדש\\nביטול = לעדכן את היום הקיים');
      if(createNew){
        forceNew=true;
      }
    }
  }

  if(forceNew){
    log.id=uid();
    log.created_at=new Date().toISOString();
    log.updated_at=new Date().toISOString();
    log.sort_order=logs.length+1;
    editingId=null;
  }

  if(!fromAuto) pushUndo(forceNew?'יצירת יום עבודה חדש':'שמירת יום עבודה');

  const i=logs.findIndex(x=>x.id===log.id);
  if(i>=0) logs[i]=log; else logs.unshift(log);

  addProject(log.project_number,log.project_name);addContractorToMemory(log.contractor_name);
  saveLocal();

  let cloud=false;
  if(accessToken) cloud=await cloudUpsert(log);

  if(cloud){storageMode='cloud';setStatus('☁️ נשמר בענן Supabase')}
  else{storageMode='local';setStatus('💾 נשמר מקומית')}

  editingId=log.id;
  renderProjectFilter();renderContractorFilter();renderContractorManager();renderContractorSelectOptions(log.contractor_name);renderAll();

  if(fromAuto){setAutoSaveText(cloud?'נשמר בענן אוטומטית':'נשמר מקומית אוטומטית')}
  else if(!silent) toast('יום העבודה נשמר ביומן ✅',cloud?'ok':'warn')
}

function filteredLogs(){
  const q=$('searchInput').value.trim().toLowerCase(),pf=$('projectFilter').value,cf=$('contractorFilter').value,sort=$('sortMode').value;
  let out=logs.filter(l=>{
    if(pf!=='all'&&l.project_number!==pf)return false;
    if(!monthInRange(l.month_key))return false;
    if(cf!=='all'&&l.contractor_name!==cf)return false;
    if(q){
      const blob=[l.project_number,l.project_name,l.contractor_name,l.contractor_manager,l.tower,l.floor,l.area,l.room,l.planned_work,l.done_work,l.quantities,l.notes,l.next_tasks,(l.workers||[]).map(w=>w.name).join(' '),(l.tasks||[]).map(t=>t.task).join(' ')].join(' ').toLowerCase();
      if(!blob.includes(q))return false
    }
    return true
  });
  if(sort==='manual')out.sort((a,b)=>Number(a.sort_order||0)-Number(b.sort_order||0));
  if(sort==='date_desc')out.sort((a,b)=>(b.log_date||'').localeCompare(a.log_date||''));
  if(sort==='date_asc')out.sort((a,b)=>(a.log_date||'').localeCompare(b.log_date||''));
  if(sort==='contractor')out.sort((a,b)=>(a.contractor_name||'').localeCompare(b.contractor_name||'','he'));
  return out
}

function logsForAllContractors(){
  const pf=$('projectFilter').value, q=$('searchInput').value.trim().toLowerCase();
  return logs.filter(l=>{
    if(pf!=='all'&&l.project_number!==pf) return false;
    if(!monthInRange(l.month_key)) return false;
    if(q){
      const blob=[l.project_number,l.project_name,l.contractor_name,l.contractor_manager,l.tower,l.floor,l.area,l.room,l.planned_work,l.done_work,l.quantities,l.notes,l.next_tasks,(l.workers||[]).map(w=>w.name).join(' '),(l.tasks||[]).map(t=>t.task).join(' ')].join(' ').toLowerCase();
      if(!blob.includes(q)) return false;
    }
    return true;
  }).sort((a,b)=>(a.contractor_name||'').localeCompare(b.contractor_name||'','he') || (a.log_date||'').localeCompare(b.log_date||''));
